local function checkAntiCheat()
    if player.Character and player.Character:FindFirstChild("Humanoid") then
        local humanoid = player.Character.Humanoid
        if humanoid.WalkSpeed > 50 or humanoid.JumpPower > 100 then
            return false, "T·ªëc ƒë·ªô ho·∫∑c nh·∫£y b·∫•t th∆∞·ªùng!"
        end
        if (player.Character.HumanoidRootPart.Position - game.Workspace:GetPivot().Position).Magnitude > 5000 then
            return false, "Di chuy·ªÉn qu√° xa!"
        end
    end
    return true, ""
end

local dragging, dragStart, startPos
frame.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = true
        dragStart = input.Position
        startPos = frame.Position
        input.Changed:Connect(function()
            if input.UserInputState == Enum.UserInputState.End then
                dragging = false
            end
        end)
    end
end)

frame.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseMovement and dragging then
        local delta = input.Position - dragStart
        frame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
    end
end)
